# Application Configuration
VITE_APP_TITLE=Admin System

VITE_BASE_URL=/markdown-preview-demo

# Router Configuration
VITE_ROUTER_HISTORY_MODE=history

VITE_ROUTE_HOME=home

# Icon Configuration
VITE_ICON_PREFIX=icon
VITE_ICON_LOCAL_PREFIX=local-icon
VITE_MENU_ICON=mdi:menu

# Backend Service Configuration
VITE_SERVICE_BASE_URL=http://localhost:3000

# ==================== 业务状态码配置 ====================
# 注意：以下配置只能使用业务错误码（code 字段），不能使用 HTTP 状态码

# 成功状态码（逗号分隔，无空格）
# 200 = SUCCESS (通用成功)
# 201 = CREATED (创建成功)
VITE_SERVICE_SUCCESS_CODE=200,201

# Token 过期但可刷新的错误码（逗号分隔，无空格）
# 1202 = TOKEN_EXPIRED (Access token has expired)
# 当收到此错误码时，系统会自动调用 /api/admin/auth/refresh 刷新 token
# 刷新成功后会重新发送原请求，刷新失败则登出
VITE_SERVICE_EXPIRED_TOKEN_CODES=1202

# 需要立即登出的错误码（逗号分隔，无空格）
# 1200 = AUTHENTICATION_FAILED (认证失败)
# 1201 = INVALID_CREDENTIALS (凭证无效)
# 1203 = TOKEN_INVALID (Token无效)
# 1204 = TOKEN_REVOKED (Token已撤销)
# 1205 = SESSION_EXPIRED (会话已过期)
# 注意：不包括 1202，因为 1202 会尝试刷新 token
VITE_SERVICE_LOGOUT_CODES=1200,1201,1203,1204,1205

# 需要显示模态框提示的登出错误码（逗号分隔，无空格）
# 1206 = ACCOUNT_LOCKED (账号已锁定)
# 1207 = ACCOUNT_DISABLED (账号已禁用)
# 2000 = ACCOUNT_BLACKLISTED (账号已列入黑名单)
# 这些错误会先显示模态框告知原因，用户点击确认后再执行登出
VITE_SERVICE_MODAL_LOGOUT_CODES=1206,1207,2000

# ==================== 其他配置 ====================
